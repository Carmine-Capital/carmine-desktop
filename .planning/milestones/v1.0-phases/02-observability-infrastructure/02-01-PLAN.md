    3. **PinStore** (`crates/carminedesktop-cache/src/pin_store.rs`) — Add health computation method:

    ```rust
    /// Compute on-demand health status for all non-expired pins.
    ///
    /// For each pinned folder:
    /// - Count total files in the pinned subtree (from items table, where folder = NULL meaning files)
    /// - Count how many of those files have an entry in cache_entries
    /// - If cached_files == total_files -> "downloaded"
    /// - If cached_files < total_files -> "partial"
    /// - "stale" is set externally via the stale_pins set (not computed here)
    ///
    /// Returns Vec of (PinnedFolder, total_files, cached_files).
    pub fn health(&self, stale_pins: &std::collections::HashSet<(String, String)>) -> carminedesktop_core::Result<Vec<(PinnedFolder, usize, usize)>> {
        let conn = self.conn.lock().map_err(|e| {
            carminedesktop_core::Error::Cache(format!("pin store lock failed: {e}"))
        })?;

        // Get all non-expired pins
        let mut pin_stmt = conn.prepare(
            "SELECT drive_id, item_id, pinned_at, expires_at FROM pinned_folders WHERE expires_at > datetime('now')"
        ).map_err(|e| carminedesktop_core::Error::Cache(format!("pin store prepare failed: {e}")))?;

        let pins: Vec<PinnedFolder> = pin_stmt.query_map([], |row| {
            Ok(PinnedFolder {
                drive_id: row.get(0)?,
                item_id: row.get(1)?,
                pinned_at: row.get(2)?,
                expires_at: row.get(3)?,
            })
        }).map_err(|e| carminedesktop_core::Error::Cache(format!("pin query failed: {e}")))?
          .filter_map(|r| r.ok())
          .collect();

        let mut results = Vec::with_capacity(pins.len());

        for pin in pins {
            // Count total files in pinned subtree: walk from the pinned item_id through
            // the items table parent chain. Files are items where folder_child_count IS NULL.
            // Use a recursive CTE to find all descendants of the pinned folder.
            let total_files: usize = conn.query_row(
                "WITH RECURSIVE subtree(inode) AS (
                    SELECT inode FROM items WHERE item_id = ?1
                    UNION ALL
                    SELECT i.inode FROM items i JOIN subtree s ON i.parent_inode = s.inode
                )
                SELECT COUNT(*) FROM items
                WHERE inode IN (SELECT inode FROM subtree)
                AND folder_child_count IS NULL
                AND item_id != ?1",
                params![pin.item_id],
                |row| row.get::<_, i64>(0),
            ).unwrap_or(0) as usize;

            // Count cached files: intersect subtree files with cache_entries
            let cached_files: usize = conn.query_row(
                "WITH RECURSIVE subtree(inode, item_id_col) AS (
                    SELECT inode, item_id FROM items WHERE item_id = ?1
                    UNION ALL
                    SELECT i.inode, i.item_id FROM items i JOIN subtree s ON i.parent_inode = s.inode
                )
                SELECT COUNT(*) FROM items
                WHERE inode IN (SELECT inode FROM subtree)
                AND folder_child_count IS NULL
                AND item_id != ?1
                AND EXISTS (
                    SELECT 1 FROM cache_entries ce
                    WHERE ce.item_id = items.item_id AND ce.drive_id = ?2
                )",
                params![pin.item_id, pin.drive_id],
                |row| row.get::<_, i64>(0),
            ).unwrap_or(0) as usize;

            results.push((pin, total_files, cached_files));
        }

        Ok(results)
    }
    ```

    NOTE on the recursive CTE: The `items` table (created by SqliteStore) has columns `inode`, `item_id`, `parent_inode`, `name`, `folder_child_count` (NULL for files). The `cache_entries` table (created by DiskCache) has columns `drive_id`, `item_id`, `file_size`, `last_accessed`. Both are in the same SQLite database. PinStore opens a separate connection to the same database, so CTE joins across both tables work.
